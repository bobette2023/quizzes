let currentQuestion = 0;
let scores = {Porous:0, Rigid:0, Healthy:0};
function updateProgress() {
    const total = questions.length;
    const answered = currentQuestion;
    const percent = (answered / total) * 100;
    document.getElementById("progressText").innerText =
        answered < total
        ? "Question " + (answered + 1) + " of " + total
        : "Assessment Complete";
    document.getElementById("progressBar").style.width = percent + "%";
}
function loadQuestion() {
    updateProgress();
    const quizDiv = document.getElementById("quiz");
    quizDiv.innerHTML = "";
    if (currentQuestion < questions.length) {
        const qObj = questions[currentQuestion];
        const questionEl = document.createElement("div");
        questionEl.className = "question";
        questionEl.innerHTML = "<strong>" + qObj.q + "</strong>";
        const answersDiv = document.createElement("div");
        answersDiv.className = "answers";
        qObj.a.forEach((answer, index) => {
            const label = document.createElement("label");
            label.innerHTML =
                `<input type="radio" name="answer" value="${qObj.score[index]}"> ${answer}`;
            answersDiv.appendChild(label);
        });
        quizDiv.appendChild(questionEl);
        quizDiv.appendChild(answersDiv);
    } else {
        showResults();
    }
}
function showResults() {
    document.getElementById("nextBtn").style.display = "none";
    updateProgress();
    const resultDiv = document.getElementById("result");
    resultDiv.style.display = "block";
    let maxScore = Math.max(scores.Porous, scores.Rigid, scores.Healthy);
    let types = Object.keys(scores).filter(type => scores[type] === maxScore);
    if (types.length === 1) {
        resultDiv.innerHTML =
        "<h3>Your dominant boundary style: " + types[0] + "</h3>" +
        "<p>Porous = Difficulty saying no and overextending yourself.<br>" +
        "Rigid = Strong walls and limited vulnerability.<br>" +
        "Healthy = Clear, balanced, respectful boundaries.</p>";
    } else {
        resultDiv.innerHTML =
        "<h3>You show a mix of: " + types.join(" & ") + "</h3>" +
        "<p>Your boundaries may shift depending on the situation.</p>";
    }
    resultDiv.innerHTML += "<br><button onclick='restartQuiz()'>Restart</button>";
}
document.getElementById("nextBtn").addEventListener("click", function() {
    const selected = document.querySelector('input[name="answer"]:checked');
    if (!selected) {
        alert("Please select an answer before continuing.");
        return;
    }
    scores[selected.value]++;
    currentQuestion++;
    loadQuestion();
});
function restartQuiz() {
    currentQuestion = 0;
    scores = {Porous:0, Rigid:0, Healthy:0};
    document.getElementById("nextBtn").style.display = "inline-block";
    document.getElementById("result").style.display = "none";
    loadQuestion();
}
